import pytest
from datetime import datetime, timedelta

**Data seeding fixture — `db_with_realistic_data`:**
Seeds 20 PortfolioSlots with ea_ids 1000-1019, buy_price ranging 20000-80000, sell_price = buy_price * 1.3. For each slot, inserts 2 TradeRecords:
- Record 1: action_type="buy", outcome="bought", price=buy_price, recorded_at=now - 2h
- Record 2: action_type="list", outcome="listed", price=sell_price, recorded_at=now - 1h

This creates a realistic "all players listed on market" state with 40 total TradeRecords.

```python
@pytest.fixture
async def db_with_realistic_data():
    engine, session_factory = await create_engine_and_tables("sqlite+aiosqlite:///:memory:")
    now = datetime.utcnow()
    async with session_factory() as session:
        for i in range(20):
            ea_id = 1000 + i
            buy_price = 20000 + i * 3000
            sell_price = int(buy_price * 1.3)
            session.add(PortfolioSlot(
                ea_id=ea_id, buy_price=buy_price, sell_price=sell_price,
                added_at=now - timedelta(hours=3),
            ))
            session.add(TradeRecord(
                ea_id=ea_id, action_type="buy", price=buy_price,
                outcome="bought", recorded_at=now - timedelta(hours=2),
            ))
            session.add(TradeRecord(
                ea_id=ea_id, action_type="list", price=sell_price,
                outcome="listed", recorded_at=now - timedelta(hours=1),
            ))
        await session.commit()
    yield engine, session_factory
    await engine.dispose()
```

**Timing pattern (critical — warmup before measurement):**
```python
async def test_pending_action_latency(db_with_realistic_data):
    """GET /actions/pending responds in < 50ms with 20-slot portfolio (in-memory SQLite)."""
    _, session_factory = db_with_realistic_data
    app = make_full_test_app(session_factory)
    async with AsyncClient(transport=ASGITransport(app=app), base_url="http://test") as client:
        # Warmup — first request may be slower due to connection pool init
        await client.get("/api/v1/actions/pending")
        # Timed run
        start = time.perf_counter()
        resp = await client.get("/api/v1/actions/pending")
        elapsed_ms = (time.perf_counter() - start) * 1000
    assert resp.status_code == 200
    assert elapsed_ms < 50, f"GET /actions/pending took {elapsed_ms:.1f}ms — threshold is 50ms"
```
