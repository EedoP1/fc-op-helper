       ```python
       import subprocess, sys, time, httpx
       env = {**os.environ, "TEST_DB_PATH": str(test_db_path)}
       proc = subprocess.Popen(
           [sys.executable, "-m", "uvicorn",
            "tests.integration.server_harness:app",
            "--host", "127.0.0.1",
            "--port", str(server_port),
            "--no-access-log"],
           env=env,
           stdout=subprocess.PIPE,
           stderr=subprocess.PIPE,
       )
       # Synchronous readiness poll — no async, no event loop issues
       base_url = f"http://127.0.0.1:{server_port}"
       for _ in range(50):  # 5 seconds max
           try:
               r = httpx.get(f"{base_url}/api/v1/health", timeout=0.5)
               if r.status_code == 200:
                   break
           except (httpx.ConnectError, httpx.ReadTimeout):
               pass
           time.sleep(0.1)
       else:
           proc.kill()
           stdout, stderr = proc.communicate(timeout=5)
           raise RuntimeError(
               f"Test server failed to start\nstdout: {stdout.decode()}\nstderr: {stderr.decode()}"
           )
       yield proc
       proc.terminate()
       proc.wait(timeout=5)
       ```
