    import { waitForElement, clickElement, randomDelay, clearAndType } from './dom-utils';
    import { SELECTORS } from '../selectors';
    import type { PendingAction } from '../storage';
    import type { ActionOutcome } from '../messages';

    /**
     * Find the target card in unassigned items by player name.
     * Iterates visible card elements and matches text content.
     * Returns the matching card element, or throws if not found.
     */
    async function findCardByName(playerName: string): Promise<Element> {
      await waitForElement(SELECTORS.unassigned.playerCard);
      const cards = document.querySelectorAll(SELECTORS.unassigned.playerCard);
      for (const card of cards) {
        const nameEl = card.querySelector(SELECTORS.unassigned.playerName);
        if (nameEl && nameEl.textContent?.trim().toLowerCase().includes(playerName.toLowerCase())) {
          return card;
        }
      }
      const err = new Error(`Card not found for player: ${playerName}`) as any;
      err.selector = SELECTORS.unassigned.playerCard;
      throw err;
    }

    /**
     * Execute the LIST flow for a pending action.
     * Per D-05: navigate to unassigned items, find the purchased card, list at OP sell price.
     * target_price for LIST actions is the OP sell price from the portfolio.
     */
    export async function executeList(action: PendingAction): Promise<ActionOutcome> {
      // Navigate to unassigned items (per D-01, D-05)
      await clickElement(SELECTORS.unassigned.navButton);
      await randomDelay();

      // Find the card we just bought
      const card = await findCardByName(action.player_name);
      (card as HTMLElement).click();
      await randomDelay();

      // Click "list on market" button
      await clickElement(SELECTORS.unassigned.listOnMarketButton);
      await randomDelay();

      // Set listing prices
      // Start price slightly below BIN to attract bid attention
      const startPrice = Math.max(action.target_price - 100, 150);
      await clearAndType(SELECTORS.listing.startPriceInput, String(startPrice));
      await clearAndType(SELECTORS.listing.buyNowPriceInput, String(action.target_price));

      // Click list button to confirm
      await clickElement(SELECTORS.listing.listButton);
      await randomDelay();

      return { status: 'success', price: action.target_price, result: 'listed' };
    }
    ```

    import { waitForElement, clickElement, randomDelay } from './dom-utils';
    import { SELECTORS } from '../selectors';
    import type { ActionOutcome } from '../messages';

    /**
     * Execute the RELIST flow.
     * Per D-06: navigate to Transfer List, click EA's built-in "Relist All" button.
     * This relists ALL expired cards at their previous list price — not just portfolio cards.
     * Price is 0 in the outcome because Relist All uses the previous listing price,
     * which we don't parse from the DOM (the backend already knows the price).
     */
    export async function executeRelist(): Promise<ActionOutcome> {
      // Navigate to Transfer List (per D-01)
      await clickElement(SELECTORS.nav.transferList);
      await randomDelay();

      // Wait for the transfer list page to load
      await waitForElement(SELECTORS.transferList.relistAllButton);

      // Click Relist All button (per D-06)
      await clickElement(SELECTORS.transferList.relistAllButton);
      await randomDelay();

      return { status: 'success', price: 0, result: 'relisted' };
    }
    ```
