    ```typescript
    import { waitForElement, clickElement, randomDelay, typeIntoInput, clearAndType } from './dom-utils';
    import { SELECTORS } from '../selectors';
    import type { PendingAction } from '../storage';
    import type { ActionOutcome } from '../messages';

    const SWEEP_STEP = 200;    // coin step for price sweep
    const MAX_SWEEP_ITERATIONS = 20;

    /** Navigate to transfer market search page via sidebar nav. Per D-01. */
    async function navigateToSearch(): Promise<void> {
      await clickElement(SELECTORS.nav.transferMarket);
      await waitForElement(SELECTORS.transferMarket.playerNameInput);
    }

    /** Fill search form with player details. Per D-02. */
    async function fillSearchForm(action: PendingAction, maxBin: number): Promise<void> {
      await typeIntoInput(SELECTORS.transferMarket.playerNameInput, action.player_name);
      await randomDelay(400, 800);
      // Wait for autocomplete suggestion and select first result
      await clickElement(SELECTORS.transferMarket.playerNameInput);
      await randomDelay(300, 600);
      // Set max BIN price
      await clearAndType(SELECTORS.transferMarket.maxBinInput, String(maxBin));
    }

    /** Check if search results page has any listings. */
    async function hasResults(): Promise<boolean> {
      try {
        await waitForElement(SELECTORS.searchResults.listingRow, 3000);
        return true;
      } catch {
        // Check for explicit "no results" message
        try {
          await waitForElement(SELECTORS.searchResults.noResults, 1000);
          return false;
        } catch {
          return false;
        }
      }
    }

    /** Read the BIN price from the first (cheapest) listing. */
    async function readCheapestBin(): Promise<number> {
      const el = await waitForElement(SELECTORS.searchResults.binPrice);
      const text = el.textContent?.replace(/[^0-9]/g, '') ?? '';
      const price = parseInt(text, 10);
      if (isNaN(price) || price <= 0) {
        throw Object.assign(
          new Error(`Invalid BIN price text: "${el.textContent}"`),
          { selector: SELECTORS.searchResults.binPrice },
        );
      }
      return price;
    }

    /** Go back to search page from results. */
    async function goBackToSearch(): Promise<void> {
      await clickElement(SELECTORS.searchResults.backButton);
      await waitForElement(SELECTORS.transferMarket.searchButton);
    }

    /**
     * Adaptive price sweep per D-03:
     * Phase 1: If no results at target_price, increase maxBin until results appear.
     * Phase 2: Once results exist, decrease maxBin to isolate cheapest listing.
     * Returns the cheapest BIN price found, or null if no listings exist at all.
     */
    async function sweepForCheapestListing(
      action: PendingAction,
    ): Promise<number | null> {
      let maxBin = action.target_price;

      // Phase 1: ensure any results exist by raising max BIN
      let foundResults = false;
      for (let i = 0; i < MAX_SWEEP_ITERATIONS; i++) {
        await fillSearchForm(action, maxBin);
        await clickElement(SELECTORS.transferMarket.searchButton);
        await randomDelay();
        if (await hasResults()) {
          foundResults = true;
          break;
        }
        await goBackToSearch();
        maxBin += SWEEP_STEP;
      }

      if (!foundResults) return null;

      // Read the cheapest BIN at current maxBin
      let cheapestBin = await readCheapestBin();

      // Phase 2: narrow down — lower maxBin until no results, then the last successful maxBin had the cheapest
      let lastGoodBin = cheapestBin;
      for (let i = 0; i < MAX_SWEEP_ITERATIONS; i++) {
        const trial = cheapestBin - SWEEP_STEP;
        if (trial <= 0) break;

        await goBackToSearch();
        await fillSearchForm(action, trial);
        await clickElement(SELECTORS.transferMarket.searchButton);
        await randomDelay();

        if (!await hasResults()) break;

        cheapestBin = await readCheapestBin();
        lastGoodBin = cheapestBin;
      }

      return lastGoodBin;
    }

    /** Click Buy Now and confirm the purchase dialog. */
    async function executeBuyNow(): Promise<void> {
      await clickElement(SELECTORS.searchResults.firstListing);
      await randomDelay();
      await clickElement(SELECTORS.searchResults.buyNowButton);
      await waitForElement(SELECTORS.searchResults.confirmDialog);
      await clickElement(SELECTORS.searchResults.confirmButton);
      await randomDelay();
    }

    /**
     * Execute the full BUY flow for a pending action.
     * Per D-01: navigates to transfer market autonomously.
     * Per D-02: searches by player name and max BIN.
     * Per D-03: adaptive price sweep to find cheapest.
     * Per D-04: price guard — skip if cheapest BIN > target_price.
     */
    export async function executeBuy(action: PendingAction): Promise<ActionOutcome> {
      await navigateToSearch();

      const cheapestBin = await sweepForCheapestListing(action);

      if (cheapestBin === null) {
        return { status: 'skipped', reason: 'price_above_guard' };
      }

      // D-04: Price guard — only buy if at or below target
      if (cheapestBin > action.target_price) {
        return { status: 'skipped', reason: 'price_above_guard' };
      }

      await executeBuyNow();

      return { status: 'success', price: cheapestBin, result: 'bought' };
    }
    ```
