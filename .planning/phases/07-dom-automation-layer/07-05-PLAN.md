    ```typescript
     * Dispatch to the correct automation flow based on action_type.
     * Returns an ActionOutcome. Catches errors and wraps in error outcome.

    async function runAutomation(action: PendingAction): Promise<ActionOutcome> {
      try {
        switch (action.action_type) {
          case 'BUY':
            return await executeBuy(action);
          case 'LIST':
            return await executeList(action);
          case 'RELIST':
            return await executeRelist();
          default:
            return {
              status: 'error',
              selector: 'unknown',
              message: `Unknown action type: ${action.action_type}`,
            };
        }
      } catch (err: any) {
        // D-10, AUTO-07: wrap DOM failures into error ActionOutcome
        return {
          status: 'error',
          selector: err.selector ?? 'unknown',
          message: err.message ?? String(err),
        };
      }
    }
    ```

    1. Replace the current `maybePoll()` function to: after storing the pending action, send EXECUTE_ACTION to the content script (instead of just PING), receive ACTION_RESULT, and call POST /complete:

    ```typescript
    async function maybePoll(): Promise<void> {
      const enabled = await enabledItem.getValue();
      if (!enabled) return;

      try {
        const res = await fetch(`${BACKEND_URL}/api/v1/actions/pending`);
        if (!res.ok) return;
        const data = await res.json();
        if (!data.action) return;

        await lastActionItem.setValue(data.action);

        // Send EXECUTE_ACTION to content script and await ACTION_RESULT (per D-07)
        const result = await sendExecuteAction(data.action);

        if (result && result.type === 'ACTION_RESULT') {
          await reportOutcome(data.action.id, result.outcome);
        }
      } catch (e) {
        console.error('[OP Seller] poll failed:', e);
      }
    }
    ```

    2. Add `sendExecuteAction` helper:
    ```typescript
    /**
     * Send EXECUTE_ACTION to the active EA Web App tab.
     * Returns the ACTION_RESULT response, or null if content script not reachable.
     * Wraps in try/catch — service worker must never throw (Pitfall 1).
     */
    async function sendExecuteAction(action: PendingAction): Promise<ExtensionMessage | null> {
      try {
        const [tab] = await chrome.tabs.query({
          url: 'https://www.ea.com/ea-sports-fc/ultimate-team/web-app/*',
          active: true,
        });
        if (!tab?.id) return null;
        return await chrome.tabs.sendMessage(tab.id, {
          type: 'EXECUTE_ACTION',
          action,
        } satisfies ExtensionMessage);
      } catch {
        return null;
      }
    }
    ```

    3. Add `reportOutcome` helper:
    ```typescript
    /**
     * Report action outcome to backend via POST /actions/{id}/complete.
     * Per D-11: report immediately, don't rely on 5-minute stale timeout.
     */
    async function reportOutcome(actionId: number, outcome: ActionOutcome): Promise<void> {
      try {
        let price = 0;
        let outcomeStr = 'error';

        if (outcome.status === 'success') {
          price = outcome.price;
          outcomeStr = outcome.result;  // 'bought' | 'listed' | 'relisted'
        } else if (outcome.status === 'skipped') {
          outcomeStr = 'skipped';
        } else if (outcome.status === 'error') {
          outcomeStr = 'error';
          console.error(`[OP Seller] Action ${actionId} failed: ${outcome.selector} — ${outcome.message}`);
        }

        await fetch(`${BACKEND_URL}/api/v1/actions/${actionId}/complete`, {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ price, outcome: outcomeStr }),
        });
      } catch (e) {
        console.error('[OP Seller] Failed to report outcome:', e);
      }
    }
    ```
