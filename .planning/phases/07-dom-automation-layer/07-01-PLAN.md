    **Create `extension/src/automation/dom-utils.ts`** (per D-10, D-13, AUTO-05, AUTO-07):

    ```typescript
    import { SELECTORS } from '../selectors';

    let lastDelay = -1;

    /** Random delay between min and max ms. No two consecutive calls produce the same value (per D-13). */
    export async function randomDelay(minMs = 800, maxMs = 2500): Promise<number> {
      let delay: number;
      do {
        delay = Math.floor(Math.random() * (maxMs - minMs + 1)) + minMs;
      } while (delay === lastDelay);
      lastDelay = delay;
      return new Promise(resolve => setTimeout(resolve, delay, delay));
    }

    /**
     * Wait for selector to appear in DOM, polling at 200ms intervals.
     * Per D-10: wait up to timeoutMs, then throw with selector name (AUTO-07).
     */
    export async function waitForElement(
      selector: string,
      timeoutMs = 5000,
      pollIntervalMs = 200,
    ): Promise<Element> {
      const deadline = Date.now() + timeoutMs;
      while (Date.now() < deadline) {
        const el = document.querySelector(selector);
        if (el) return el;
        await new Promise(r => setTimeout(r, pollIntervalMs));
      }
      const err = new Error(`Element not found: ${selector}`) as any;
      err.selector = selector;
      throw err;
    }

    /** Click element after randomized delay (human pacing per AUTO-05). */
    export async function clickElement(selector: string): Promise<void> {
      const el = await waitForElement(selector);
      await randomDelay();
      (el as HTMLElement).click();
    }

    /** Type a value into an input field — focus, clear, set value, dispatch input event. */
    export async function typeIntoInput(selector: string, value: string): Promise<void> {
      const el = await waitForElement(selector) as HTMLInputElement;
      await randomDelay();
      el.focus();
      el.value = '';
      el.value = value;
      el.dispatchEvent(new Event('input', { bubbles: true }));
      el.dispatchEvent(new Event('change', { bubbles: true }));
    }

    /** Clear an input field then type a new value. */
    export async function clearAndType(selector: string, value: string): Promise<void> {
      const el = await waitForElement(selector) as HTMLInputElement;
      await randomDelay();
      el.focus();
      el.value = '';
      el.dispatchEvent(new Event('input', { bubbles: true }));
      await randomDelay(300, 600);
      el.value = value;
      el.dispatchEvent(new Event('input', { bubbles: true }));
      el.dispatchEvent(new Event('change', { bubbles: true }));
    }
    ```
