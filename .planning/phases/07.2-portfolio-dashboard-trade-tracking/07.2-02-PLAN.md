    ```typescript
    /**
     * Fetch portfolio status from backend for the dashboard panel.
     * Returns per-player trade status, cumulative stats, and profit summary.
     */
    async function handleDashboardStatus(): Promise<ExtensionMessage> {
      try {
        const resp = await fetch(`${BACKEND_URL}/api/v1/portfolio/status`);
        if (!resp.ok) {
          return { type: 'DASHBOARD_STATUS_RESULT', data: null, error: `Backend returned ${resp.status}` };
        }
        const data = await resp.json();
        return { type: 'DASHBOARD_STATUS_RESULT', data, error: undefined };
      } catch (err) {
        const message = err instanceof Error ? err.message : 'Unknown error';
        return { type: 'DASHBOARD_STATUS_RESULT', data: null, error: message };
      }
    }
    ```

    Import `DashboardData` is NOT needed in background.ts — the handler returns the raw JSON as `data` and the type is inferred from the `ExtensionMessage` union.

    ```typescript
      const contentArea = container.querySelector('.op-seller-tab-content');
      if (contentArea) contentArea.innerHTML = '';
      const content = contentArea || container;

      // Loading state

      chrome.runtime.sendMessage({ type: 'DASHBOARD_STATUS_REQUEST' } satisfies ExtensionMessage)
        .then((res: ExtensionMessage) => {
          if (res.type === 'DASHBOARD_STATUS_RESULT') {
            content.innerHTML = '';
            if (res.error || !res.data) {
              const errEl = document.createElement('div');
              errEl.textContent = `Error: ${res.error || 'No data'}`;
              errEl.style.color = '#f44';
              content.appendChild(errEl);
              return;
            }
            renderDashboardContent(content as HTMLElement, res.data);
          }
        })
        .catch((err: Error) => {
          content.innerHTML = '';
          const errEl = document.createElement('div');
          errEl.textContent = `Error: ${err.message}`;
          errEl.style.color = '#f44';
    ```
